Big O

Determine the Big O (worst case) runtime of the following sections of code.

Be sure to define what your basic unit of work is for each section.

// snippet 1
int n;
for (unsigned i = 0; i < n; ++i) {
    cout << n - i << endl;
}

// snippet 2
int n = 100;
vector<int> v(n);

// snippet 3
int n;
for (int i = 0; i < n; ++i) {
    for (int j = 0; j < n; ++i) {
        cout << '.';
    }
}

// snippet 4
int n;
int m;
for (int i = 0; i < n; ++i) {
    for (int j = 0; j < m; ++i) {
        cout << '.';
    }
}

// snippet 5
int n;
for (int i = 1; i < n; i *= 2) {
    cout << '.' << endl;
}
